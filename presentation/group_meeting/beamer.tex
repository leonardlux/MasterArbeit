% Slide: Error Equations
\begin{frame}
    \begin{block}{$X$-Errors}
        \begin{equation}
            \varepsilon_{X}(\rho) = (1-p) \rho + p \cdot X \rho X
        \end{equation}
    \end{block}
    
    % Depo 1
    \begin{block}{Depolarizing Errors}
        \begin{equation}
            \varepsilon_{D_1}(\rho) = (1-p) \rho + \frac{p}{3} \sum^3_{i=1} E_i \rho E_i
        \end{equation}
        \begin{equation*}
            E_i \in \{ X, Y, Z \}
        \end{equation*}

    % Depo 2
        \begin{equation}
            \varepsilon_{D_2}(\rho) = (1-p) \rho + \frac{p}{15} \sum_{i=1}^{15} E_i \rho E_i,
        \end{equation}
        \begin{equation*}
            E_i \in \{E_k \otimes E_l | E_k,E_l \in \{\mathbb{1},X,Y,Z\} \} \setminus \{\mathbb{1}\otimes\mathbb{1} \}
        \end{equation*}
    \end{block}
\end{frame}

\section{MLD Error Model} 
\begin{frame}{Reorder}
    \begin{itemize}
        \item commute operators 
        \item $tr_q$ 
        \item $Z \simeq I$, $Y \simeq X$
    \end{itemize}
        \begin{quantikz}
            \lstick{$\ket{\Psi_A}$}&\ctrl{1}&\targ{}&\gate[]{Trash}\\
            \lstick{$\ket{\Psi_B}$}&\targ{}&&&\gate[1,style={noisy}]{X_2}\wire[d][1]{q}&\gate[1,style={noisy},]{X}&\meter{$Z$}&\setwiretype{c}\\
            \lstick{$\ket{\Psi_C}$}&&\ctrl{-2}&\gate[]{H}&\gate[1,style={noisy}]{X_2}&\gate[1,style={noisy},]{X}&\meter{$Z$}&\setwiretype{c}\\
        \end{quantikz}
    \begin{itemize}
        \item 3 distinct channels 
        \item mapping depends on ordering
    \end{itemize}
\end{frame}

\begin{frame}{Channel Equations}
    \begin{block}{Bit-flip channel}
        \begin{equation*}
        \varepsilon_{X,B/C}(\rho) = (1-p) \hat{I}+ p_x \cdot \hat{X}_{B/C} 
        \end{equation*}
    \end{block}
    \begin{block}{$X_2$ channel}
        \begin{equation*}
        \varepsilon_{X_2,BC}(\rho) = (1-p_D) \hat{I} + \frac{1}{3} p_D \cdot \left(\hat{X}_B + \hat{X}_C + \hat{X}_B\otimes\hat{X}_C\right) 
        \end{equation*}
    \end{block}

    \begin{itemize}
        \item relabel to MLD perspective: 
        \begin{itemize}
            \item $p_x$ on system $B$: $p_x$
            \item $p_x$ on system $C$: $p_z$ 
        \end{itemize}
    \end{itemize}
\end{frame}

\begin{frame}{MLD Error Model}
    \begin{block}{Resulting Probabilities}
        \begin{equation*}
            p_{(i,j)}^{(MLD)} = \frac{1}{3}p_{D} + (1- \frac{4}{3}p_D)\cdot p_{x,i} \cdot p_{z,j}
        \end{equation*}
        \vspace{0.5cm}
        \begin{equation*}
            p_{x,i} = 
            \begin{cases}
                p_x & \text{if } i=1  \\
                (1-p_x) & \text{if } i=0\\
            \end{cases}
        \end{equation*}
    \end{block}
\end{frame}
