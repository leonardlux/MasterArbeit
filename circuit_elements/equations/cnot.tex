% Trivial CNOTs (Steane Type EC)
    

\begin{equation*}
    C^{\ket{0}}NOT^{\ket{\Psi}} \left(\ket{\Psi}_L \otimes \ket{0}_L \right)  = \ket{\Psi}_L \otimes \ket{0}_L
\end{equation*}

\begin{equation*}
    C^{\ket{0}}NOT^{\ket{\Psi}} \left({\color{red} Z_i}\ket{\Psi}_L \otimes \ket{0}_L \right)  = {\color{red} Z_i}\ket{\Psi}_L \otimes {\color{red} Z_i}\ket{0}_L
\end{equation*}

\begin{equation}
    C^{\ket{\Psi}}NOT^{\ket{+}} \left(\ket{\Psi}_L \otimes \ket{+}_L \right)  = \ket{\Psi}_L \otimes \ket{+}_L
\end{equation}

\begin{equation}
    C^{\ket{\Psi}}NOT^{\ket{+}} \left({\color{red} X_i}\ket{\Psi}_L \otimes \ket{+}_L \right)  = {\color{red} X_i}\ket{\Psi}_L \otimes {\color{red} X_i}\ket{+}_L
\end{equation}
